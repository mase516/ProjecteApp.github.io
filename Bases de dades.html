Creació Taules

Taula Formació

CREATE TABLE `FORMACIÓ` (
  `ID_F` int(11) NOT NULL,
  `Nom_Usuari` varchar(25) NOT NULL,
  `Punts_Totals` int(11) DEFAULT NULL,
  `TOP` varchar(25) DEFAULT NULL,
  `MID` varchar(25) DEFAULT NULL,
  `JNG` varchar(25) DEFAULT NULL,
  `ADC` varchar(25) DEFAULT NULL,
  `SUP` varchar(25) DEFAULT NULL,
  PRIMARY KEY (`ID_F`),
  CONSTRAINT `FORMACIÓ_ibfk_1` FOREIGN KEY (`ID_F`) REFERENCES `USUARIS` (`ID`)
)

Taula Jugadors

Taula Lliga
